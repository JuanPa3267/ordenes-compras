-- 1. REINICIAR ESQUEMA
DROP SCHEMA IF EXISTS public CASCADE;
CREATE SCHEMA public AUTHORIZATION pg_database_owner;
COMMENT ON SCHEMA public IS 'standard public schema';

-- 2. EXTENSIONES (Para funciones criptográficas como gen_random_uuid, crypt, etc.)
CREATE EXTENSION IF NOT EXISTS pgcrypto;

-- 3. TIPOS PERSONALIZADOS (ENUMS)
CREATE TYPE public.estado_oc AS ENUM (
	'BORRADOR',
	'ENVIADO',
	'CANCELADO');

-- 4. TABLAS BASE (Sin dependencias de FK)
CREATE TABLE public.cargos (
	id_cargo bigserial NOT NULL,
	descripcion varchar(80) NOT NULL,
	CONSTRAINT cargos_descripcion_key UNIQUE (descripcion),
	CONSTRAINT cargos_pkey PRIMARY KEY (id_cargo)
);

CREATE TABLE public.clientes (
	id_cliente bigserial NOT NULL,
	nit varchar(30) NOT NULL,
	nombre varchar(150) NOT NULL,
	direccion varchar(200) NOT NULL,
	telefono varchar(30) NULL,
	fecha_creacion timestamptz DEFAULT now() NOT NULL,
	CONSTRAINT clientes_nit_key UNIQUE (nit),
	CONSTRAINT clientes_pkey PRIMARY KEY (id_cliente)
);

CREATE TABLE public.distribuidores (
	id_distribuidor bigserial NOT NULL,
	nombre varchar(80) NOT NULL,
	CONSTRAINT distribuidores_nombre_key UNIQUE (nombre),
	CONSTRAINT distribuidores_pkey PRIMARY KEY (id_distribuidor)
);

CREATE TABLE public.fabricantes (
	id_fabricante bigserial NOT NULL,
	nombre varchar(80) NOT NULL,
	CONSTRAINT fabricantes_nombre_key UNIQUE (nombre),
	CONSTRAINT fabricantes_pkey PRIMARY KEY (id_fabricante)
);

CREATE TABLE public.monedas (
	id_moneda bigserial NOT NULL,
	codigo varchar(10) NOT NULL,
	nombre varchar(60) NOT NULL,
	CONSTRAINT monedas_codigo_key UNIQUE (codigo),
	CONSTRAINT monedas_pkey PRIMARY KEY (id_moneda)
);

CREATE TABLE public.proveedores (
	id_proveedor int8 GENERATED BY DEFAULT AS IDENTITY NOT NULL,
	nombre varchar(150) NOT NULL,
	direccion varchar(200) NOT NULL,
	telefono varchar(30) NULL,
	correo varchar(150) NOT NULL,
	fecha_creacion timestamptz DEFAULT now() NOT NULL,
	CONSTRAINT proveedores_pkey PRIMARY KEY (id_proveedor)
);

CREATE TABLE public.tipos_item (
	id_tipo_item bigserial NOT NULL,
	nombre varchar(60) NOT NULL,
	descripcion varchar(255) NULL,
	CONSTRAINT tipos_item_nombre_key UNIQUE (nombre),
	CONSTRAINT tipos_item_pkey PRIMARY KEY (id_tipo_item)
);

-- 5. TABLAS DE SEGUNDO NIVEL (Dependen de las bases)
CREATE TABLE public.contactos_cliente (
	id_contacto_cliente bigserial NOT NULL,
	id_cliente int8 NOT NULL,
	nombre_completo varchar(120) NOT NULL,
	correo varchar(150) NOT NULL,
	telefono varchar(30) NULL,
	cargo varchar(80) NULL,
	activo bool DEFAULT true NOT NULL,
	CONSTRAINT contactos_cliente_pkey PRIMARY KEY (id_contacto_cliente),
	CONSTRAINT contactos_cliente_id_cliente_fkey FOREIGN KEY (id_cliente) REFERENCES public.clientes(id_cliente) ON DELETE CASCADE ON UPDATE CASCADE
);
CREATE INDEX ix_contactos_cliente_id_cliente ON public.contactos_cliente (id_cliente);

CREATE TABLE public.gerentes_cuenta (
	id_gerente bigserial NOT NULL,
	nombre_completo varchar(120) NOT NULL,
	correo varchar(150) NOT NULL,
	activo bool DEFAULT true NOT NULL,
	"password" text NULL,
	id_cargo int8 NOT NULL,
	CONSTRAINT gerentes_cuenta_correo_key UNIQUE (correo),
	CONSTRAINT gerentes_cuenta_pkey PRIMARY KEY (id_gerente),
	CONSTRAINT gerentes_cuenta_id_cargo_fkey FOREIGN KEY (id_cargo) REFERENCES public.cargos(id_cargo) ON DELETE RESTRICT ON UPDATE CASCADE
);

-- 6. TABLAS DE NEGOCIO (Ordenes de Compra)
CREATE TABLE public.ordenes_compra (
	id_oc bigserial NOT NULL,
	id_cliente int8 NOT NULL,
	fecha_emision timestamptz DEFAULT now() NOT NULL,
	id_moneda int8 NOT NULL,
	descripcion_proyecto text NOT NULL,
	oportunidad text NOT NULL,
	id_contacto_cliente int8 NULL,
	id_gerente int8 NOT NULL,
	contacto_mayorista varchar(150) NULL,
	forma_pago varchar(120) NOT NULL,
	estado public.estado_oc DEFAULT 'BORRADOR'::estado_oc NOT NULL,
	CONSTRAINT ordenes_compra_pkey PRIMARY KEY (id_oc),
	CONSTRAINT ordenes_compra_id_cliente_fkey FOREIGN KEY (id_cliente) REFERENCES public.clientes(id_cliente) ON DELETE RESTRICT ON UPDATE CASCADE,
	CONSTRAINT ordenes_compra_id_contacto_cliente_fkey FOREIGN KEY (id_contacto_cliente) REFERENCES public.contactos_cliente(id_contacto_cliente) ON DELETE SET NULL ON UPDATE CASCADE,
	CONSTRAINT ordenes_compra_id_gerente_fkey FOREIGN KEY (id_gerente) REFERENCES public.gerentes_cuenta(id_gerente) ON DELETE RESTRICT ON UPDATE CASCADE,
	CONSTRAINT ordenes_compra_id_moneda_fkey FOREIGN KEY (id_moneda) REFERENCES public.monedas(id_moneda) ON DELETE RESTRICT ON UPDATE CASCADE
);

CREATE TABLE public.totales_oc (
	id_totales bigserial NOT NULL,
	id_oc int8 NOT NULL,
	subtotal numeric(18, 2) DEFAULT 0 NOT NULL,
	iva_total numeric(18, 2) DEFAULT 0 NOT NULL,
	total numeric(18, 2) DEFAULT 0 NOT NULL,
	CONSTRAINT totales_oc_id_oc_key UNIQUE (id_oc),
	CONSTRAINT totales_oc_pkey PRIMARY KEY (id_totales),
	CONSTRAINT totales_oc_id_oc_fkey FOREIGN KEY (id_oc) REFERENCES public.ordenes_compra(id_oc) ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE public.auditoria_eventos (
	id_evento bigserial NOT NULL,
	fecha_evento timestamptz DEFAULT now() NOT NULL,
	id_gerente int8 NOT NULL,
	accion varchar(60) NOT NULL,
	entidad varchar(60) NOT NULL,
	entidad_id varchar(60) NOT NULL,
	detalle jsonb NULL,
	ip_origen inet NULL,
	user_agent text NULL,
	CONSTRAINT auditoria_eventos_pkey PRIMARY KEY (id_evento),
	CONSTRAINT auditoria_eventos_id_gerente_fkey FOREIGN KEY (id_gerente) REFERENCES public.gerentes_cuenta(id_gerente) ON DELETE RESTRICT ON UPDATE CASCADE
);

-- 7. DETALLE DE PROVEEDORES POR OC
CREATE TABLE public.oc_proveedores (
	id_oc_proveedor bigserial NOT NULL,
	id_oc int8 NOT NULL,
	id_proveedor int8 NOT NULL, -- Corregido de varchar a int8 para coincidir con tabla proveedores
	notas varchar(255) NULL,
	CONSTRAINT oc_proveedores_pkey PRIMARY KEY (id_oc_proveedor),
	CONSTRAINT oc_proveedores_id_oc_fkey FOREIGN KEY (id_oc) REFERENCES public.ordenes_compra(id_oc) ON DELETE CASCADE ON UPDATE CASCADE,
	CONSTRAINT oc_proveedores_id_proveedor_fkey FOREIGN KEY (id_proveedor) REFERENCES public.proveedores(id_proveedor) ON DELETE RESTRICT ON UPDATE CASCADE
);

CREATE TABLE public.totales_por_proveedor (
	id_totales_proveedor bigserial NOT NULL,
	id_oc_proveedor int8 NOT NULL,
	subtotal numeric(18, 2) DEFAULT 0 NOT NULL,
	iva_total numeric(18, 2) DEFAULT 0 NOT NULL,
	total numeric(18, 2) DEFAULT 0 NOT NULL,
	CONSTRAINT totales_por_proveedor_id_oc_proveedor_key UNIQUE (id_oc_proveedor),
	CONSTRAINT totales_por_proveedor_pkey PRIMARY KEY (id_totales_proveedor),
	CONSTRAINT totales_por_proveedor_id_oc_proveedor_fkey FOREIGN KEY (id_oc_proveedor) REFERENCES public.oc_proveedores(id_oc_proveedor) ON DELETE CASCADE ON UPDATE CASCADE
);

-- 8. ITEMS DE LA ORDEN (Nivel más bajo)
CREATE TABLE public.oc_items (
	id_oc_item bigserial NOT NULL,
	id_oc_proveedor int8 NOT NULL,
	descripcion text NOT NULL,
	id_tipo_item int8 NOT NULL,
	id_fabricante int8 NULL,
	id_distribuidor int8 NULL,
	parte varchar(120) NULL,
	modelo varchar(120) NULL,
	cantidad numeric(18, 4) DEFAULT 0 NOT NULL,
	valor_unitario numeric(18, 2) DEFAULT 0 NOT NULL,
	subtotal numeric(18, 2) DEFAULT 0 NOT NULL,
	iva bool DEFAULT false NOT NULL,
	porcentaje_iva numeric(6, 4) DEFAULT 0.1900 NOT NULL,
	valor_iva numeric(18, 2) DEFAULT 0 NOT NULL,
	total numeric(18, 2) DEFAULT 0 NOT NULL,
	CONSTRAINT oc_items_cantidad_check CHECK (cantidad >= 0),
	CONSTRAINT oc_items_pkey PRIMARY KEY (id_oc_item),
	CONSTRAINT oc_items_porcentaje_iva_check CHECK (porcentaje_iva >= 0),
	CONSTRAINT oc_items_valor_unitario_check CHECK (valor_unitario >= 0),
	CONSTRAINT oc_items_id_distribuidor_fkey FOREIGN KEY (id_distribuidor) REFERENCES public.distribuidores(id_distribuidor) ON DELETE SET NULL ON UPDATE CASCADE,
	CONSTRAINT oc_items_id_fabricante_fkey FOREIGN KEY (id_fabricante) REFERENCES public.fabricantes(id_fabricante) ON DELETE SET NULL ON UPDATE CASCADE,
	CONSTRAINT oc_items_id_oc_proveedor_fkey FOREIGN KEY (id_oc_proveedor) REFERENCES public.oc_proveedores(id_oc_proveedor) ON DELETE CASCADE ON UPDATE CASCADE,
	CONSTRAINT oc_items_id_tipo_item_fkey FOREIGN KEY (id_tipo_item) REFERENCES public.tipos_item(id_tipo_item) ON DELETE RESTRICT ON UPDATE CASCADE
);

-- 9. FUNCIONES DE LÓGICA (Deben ir antes que los Triggers)
CREATE OR REPLACE FUNCTION public.fn_oc_items_calcular_totales()
 RETURNS trigger
 LANGUAGE plpgsql
AS $function$
BEGIN
  NEW.subtotal := ROUND((NEW.cantidad * NEW.valor_unitario)::numeric, 2);
  IF NEW.iva THEN
    NEW.valor_iva := ROUND((NEW.subtotal * NEW.porcentaje_iva)::numeric, 2);
  ELSE
    NEW.valor_iva := 0;
  END IF;
  NEW.total := ROUND((NEW.subtotal + NEW.valor_iva)::numeric, 2);
  RETURN NEW;
END;
$function$;

CREATE OR REPLACE FUNCTION public.fn_recalcular_totales_completos()
 RETURNS trigger
 LANGUAGE plpgsql
AS $function$
DECLARE
    v_id_oc_proveedor BIGINT;
    v_id_oc BIGINT;
BEGIN
    IF (TG_OP = 'DELETE') THEN
        v_id_oc_proveedor := OLD.id_oc_proveedor;
    ELSE
        v_id_oc_proveedor := NEW.id_oc_proveedor;
    END IF;

    SELECT id_oc INTO v_id_oc FROM public.oc_proveedores WHERE id_oc_proveedor = v_id_oc_proveedor;

    -- Actualizar totales por PROVEEDOR
    INSERT INTO public.totales_por_proveedor (id_oc_proveedor, subtotal, iva_total, total)
    SELECT 
        v_id_oc_proveedor,
        COALESCE(SUM(subtotal), 0),
        COALESCE(SUM(valor_iva), 0),
        COALESCE(SUM(total), 0)
    FROM public.oc_items
    WHERE id_oc_proveedor = v_id_oc_proveedor
    ON CONFLICT (id_oc_proveedor) 
    DO UPDATE SET 
        subtotal = EXCLUDED.subtotal,
        iva_total = EXCLUDED.iva_total,
        total = EXCLUDED.total;

    -- Actualizar totales de la OC GLOBAL
    INSERT INTO public.totales_oc (id_oc, subtotal, iva_total, total)
    SELECT 
        v_id_oc,
        COALESCE(SUM(subtotal), 0),
        COALESCE(SUM(iva_total), 0),
        COALESCE(SUM(total), 0)
    FROM public.totales_por_proveedor tp
    JOIN public.oc_proveedores op ON tp.id_oc_proveedor = op.id_oc_proveedor
    WHERE op.id_oc = v_id_oc
    ON CONFLICT (id_oc) 
    DO UPDATE SET 
        subtotal = EXCLUDED.subtotal,
        iva_total = EXCLUDED.iva_total,
        total = EXCLUDED.total;

    RETURN NULL;
END;
$function$;

-- 10. TRIGGERS
CREATE TRIGGER trg_oc_items_calc 
BEFORE INSERT OR UPDATE ON public.oc_items 
FOR EACH ROW EXECUTE FUNCTION fn_oc_items_calcular_totales();

CREATE TRIGGER trg_oc_items_recalcular_global 
AFTER INSERT OR DELETE OR UPDATE ON public.oc_items 
FOR EACH ROW EXECUTE FUNCTION fn_recalcular_totales_completos();

-- 11. PERMISOS FINALES
GRANT ALL ON SCHEMA public TO pg_database_owner;
GRANT ALL ON SCHEMA public TO postgres;
