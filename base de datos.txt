-- DROP FUNCTION public.fn_oc_items_calcular_totales();

CREATE OR REPLACE FUNCTION public.fn_oc_items_calcular_totales()
 RETURNS trigger
 LANGUAGE plpgsql
AS $function$
BEGIN
  NEW.subtotal := ROUND((NEW.cantidad * NEW.valor_unitario)::numeric, 2);

  IF NEW.iva THEN
    NEW.valor_iva := ROUND((NEW.subtotal * NEW.porcentaje_iva)::numeric, 2);
  ELSE
    NEW.valor_iva := 0;
  END IF;

  NEW.total := ROUND((NEW.subtotal + NEW.valor_iva)::numeric, 2);
  RETURN NEW;
END;
$function$

-- DROP FUNCTION public.fn_recalcular_totales_completos();

CREATE OR REPLACE FUNCTION public.fn_recalcular_totales_completos()
 RETURNS trigger
 LANGUAGE plpgsql
AS $function$
DECLARE
    v_id_oc_proveedor BIGINT;
    v_id_oc BIGINT;
BEGIN
    -- 1. Determinar qué ID de proveedor estamos afectando
    IF (TG_OP = 'DELETE') THEN
        v_id_oc_proveedor := OLD.id_oc_proveedor;
    ELSE
        v_id_oc_proveedor := NEW.id_oc_proveedor;
    END IF;

    -- 2. Obtener el ID de la Orden de Compra (OC) principal
    SELECT id_oc INTO v_id_oc FROM public.oc_proveedores WHERE id_oc_proveedor = v_id_oc_proveedor;

    -- 3. Actualizar totales por PROVEEDOR
    INSERT INTO public.totales_por_proveedor (id_oc_proveedor, subtotal, iva_total, total)
    SELECT 
        v_id_oc_proveedor,
        COALESCE(SUM(subtotal), 0),
        COALESCE(SUM(valor_iva), 0),
        COALESCE(SUM(total), 0)
    FROM public.oc_items
    WHERE id_oc_proveedor = v_id_oc_proveedor
    ON CONFLICT (id_oc_proveedor) 
    DO UPDATE SET 
        subtotal = EXCLUDED.subtotal,
        iva_total = EXCLUDED.iva_total,
        total = EXCLUDED.total;

    -- 4. Actualizar totales de la OC GLOBAL
    INSERT INTO public.totales_oc (id_oc, subtotal, iva_total, total)
    SELECT 
        v_id_oc,
        COALESCE(SUM(subtotal), 0),
        COALESCE(SUM(iva_total), 0),
        COALESCE(SUM(total), 0)
    FROM public.totales_por_proveedor tp
    JOIN public.oc_proveedores op ON tp.id_oc_proveedor = op.id_oc_proveedor
    WHERE op.id_oc = v_id_oc
    ON CONFLICT (id_oc) 
    DO UPDATE SET 
        subtotal = EXCLUDED.subtotal,
        iva_total = EXCLUDED.iva_total,
        total = EXCLUDED.total;

    RETURN NULL;
END;
$function$
;
